class Human {
    String name;
    int age;

    Human(String name, int age) {
        this.name = name;
        this.age = age;
    }

    void eat() {
        System.out.println(name + " is eating.");
    }

    void sleep() {
        System.out.println(name + " is sleeping.");
    }
}

class Student extends Human {
    String school;
    int grade;

    Student(String name, int age, String school, int grade) {
        super(name, age);
        this.school = school;
        this.grade = grade;
    }

    void study() {
        System.out.println(name + " is studying in grade " + grade + ".");
    }

    void eat() {
        System.out.println(name + " is eating in the school cafeteria.");
    }
}

public class Inheritance {
    public static void main(String[] args) {
        Student s1 = new Student("Rahul", 16, "ABC School", 10);
        s1.eat();
        s1.study();
        s1.sleep();

        Human h1 = new Human("Anita", 30);
        h1.eat();
        h1.sleep();
    }
}
